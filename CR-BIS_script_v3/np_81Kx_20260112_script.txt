Function map_search
   start_map_num = 1

   loop $map_num 
        start_map_num = $start_map_num + 1
      map_index = $map_center_var[$start_map_num]
        start_map_num = $start_map_num + 1
      map_x = $map_center_var[$start_map_num]
        start_map_num = $start_map_num + 1
      map_y = $map_center_var[$start_map_num]

      dx = $current_X - $map_x 
      dy = $current_Y - $map_y
      dist = sqrt $dx * $dx + $dy * $dy
      if $start_map_num == 4
         dist_min = $dist 
         target_map_index = $map_index 
      elseif $dist < $dist_min 
         dist_min = $dist 
         target_map_index = $map_index
      Endif 
   EndLoop 
   echo load_map index $target_map_index 
   LoadOtherMap $target_map_index E
EndFunction 

Function load_map
   move_stage = 0
   ImageProperties  E
   map_size_x = $reportedValue1 
   map_size_y = $reportedValue2
   binning = $reportedValue3
   #echo $map_size_x $map_size_y $binning 
   crop_offset = ( 4096 / $binning  ) / 2

   StagePosToBufImagePos E 1 $current_X $current_Y center_x center_y
   crop_x0 = $center_x - $crop_offset 
   crop_x1 = $center_x + $crop_offset 
   crop_y0 = $center_y - $crop_offset 
   crop_y1 = $center_y + $crop_offset 

   if $map_size_x < $crop_x1 
      move_stage = $move_stage  + 1
   ElseIf  $map_size_y < $crop_y1
      move_stage = $move_stage  + 1
   ElseIf  $crop_x0 < 0 
      move_stage = $move_stage  + 1
   ElseIf  $crop_y0 < 0 
      move_stage = $move_stage  + 1
   Else
     echo $map_size_x $crop_x1 
     echo $map_size_y $crop_y1
   Endif 

   echo $crop_x0 $crop_x1 $crop_y0 $crop_y1 
   echo $map_size_x $map_size_y 

   if $move_stage == 0
      CropImage E $crop_x0 $crop_x1 $crop_y0 $crop_y1 
   Else
      echo do realign to nav item 
      RealignToNavItem 0 0 1000 1 1
      V
   Endif 
EndFunction

function focus_value
    target_focus =  RAND 
    target_focus = ( $max_defocus - $min_defocus ) * $target_focus + $min_defocus
    echo TargetDefocus is $target_focus
    SetTargetDefocus $target_focus
    SetAutofocusOffset 0
EndFunction 

function Z_focus
target_valuve = $target_focus

Loop 10
     Autofocus -1
     reportAutofocus
    recent_value = $reportedValue1
    move_value =  -1 * $recent_value + $target_valuve
   
    if $move_value < 0
            if $move_value > -0.2
                echo --> focus error $move_value micron
                break
            endif
    else
             if $move_value < 0.2
                echo --> focus error $move_value micron
                  break
             endif
    endif

    MoveStage 0 0 $move_value
    echo --> Z moved $move_value micron 

EndLoop
endfunction

Function next_VPP
   VPP_position_num  = $#VPP_list 
   ReportPhasePlatePos VPP_position
   Loop $VPP_position_num  VPP_index
      echo $VPP_position $VPP_list[$VPP_index]

   EndTry 
   Loop 77
      PhasePlateToNextPos
      ReportPhasePlatePos VPP_position
      if $VPP_position == $target_VPP 
         Break 
      Endif 
   EndLoop

Tiltbacklash = -3    #negative tilts will be backlashed, must be negative value
Drifttimes = 5         #maximum number of drift measurements before skipping
Driftinterval = 10     #wait time between drift measurements in seconds, one round of autofocus takes about 10 sec
#AbsoluteFocusLimits -10 10   #prevent runaway focus
#FocusChangeLimits -2 2          #prevent runaway focus
SuppressReports 
NoMessageBoxOnError
Require variable1
Try 
   SetDirectory $work_dir 
Catch 
   MakeDirectory $work_dir 
   SetDirectory $work_dir 
EndTry 

#create angle serial
angle_serial = {$start_angle }
P_align = {0 }
P_align_X = {0 }
M_align = {0 }
M_align_X = {0 }

if $data_mode == 1
  k = 1
  Loop $tilttimes
    #tilt plus1
     tiltangle = $start_angle + $step * $k
     if $tiltangle <= $max_angle
           angle_serial = {$angle_serial $tiltangle }
    Endif 
    #tilt min1
      tiltangle = $start_angle -  $step * $k
      if $tiltangle >= $min_angle
            angle_serial = {$angle_serial $tiltangle }
     Endif 
    k = $k + 1
    #tilt min2
     tiltangle = $start_angle -  $step * $k
      if $tiltangle >= $min_angle
            angle_serial = {$angle_serial $tiltangle }
     Endif 
    #tilt plus2
      tiltangle = $start_angle +  $step * $k
     if $tiltangle <= $max_angle
           angle_serial = {$angle_serial $tiltangle }
    Endif 
    k = $k + 1
  EndLoop
ElseIf $data_mode == 2 or $data_mode == 3
  #tilt plus
  k = 1
  Loop $tilttimes
     tiltangle = $start_angle + $step * $k
     if $tiltangle <= $max_angle
           angle_serial = {$angle_serial $tiltangle }
    Endif 
    k = $k + 1
  EndLoop

  #tilt min
  k = 1
  Loop $tilttimes
     tiltangle = $start_angle -  $step * $k
      if $tiltangle >= $min_angle
            angle_serial = {$angle_serial $tiltangle }
     Endif 
    k = $k + 1
  EndLoop
endif

echo $angle_serial
#---before data collection----
if $do_eucentric_fine  == 1
       Eucentricity 2
       UpdateItemZ
Endif 

if $skip_RealignToNavItem  == 2
   RealignToNavItem 0 0 100 3 0
endif
WalkUpTo $start_angle 
# not to restore the microscope state (set 0)
# use single-frame acquisition mode  (set 1)
# realigning with a new image up to 3 times (set 3) and if the starting shift is more than the criterion 100nm (set 100).
# end with  nonzero image shift (set 0)

V
Copy A H
Copy A G

GoToLowDoseArea  R
ReportMagIndex mag_index
SpecimenToISMatrix $mag_index
IS_xpx = -1 * $reportedValue1
IS_xpy = -1 * $reportedValue2
IS_ypx = -1 * $reportedValue3
IS_ypy = -1 * $reportedValue4
ReportObjectiveStigmator stigma_ori_X stigma_ori_Y 
ReportBeamTilt current_coma_ori_X current_coma_ori_Y 
ReportDefocus cur_focus
NormalizeAllLenses 
delay 2
OnStopCallFunc Tomo_group_X::reset_all

#-----data collection start------

CallFunction Tomo_group_X::TiltZero

k = 2
tilt_num = $#angle_serial - 1
plus_num = 1
minus_num = 1
Loop $tilt_num  
     tiltangle = $angle_serial[$k]
     if $angle_serial[$k] > $angle_serial[$k-1]
           #tilt plus
           CallFunction Tomo_group_X::TiltPlus
          plus_num = $plus_num + 1
     ElseIf  $angle_serial[$k] < $angle_serial[$k-1]
           #tilt min
           CallFunction Tomo_group_X::TiltMinus
           minus_num = $minus_num + 1
    Endif 
   echo P_align $P_align 
   echo M_align $M_align 
    k = $k + 1
EndLoop

Function check_autofill
  Loop 10
    AreDewarsFilling
    Filling = $reportedValue1
    If $Filling == 0 
       Break
    Endif
    Delay 2 min
  EndLoop
EndFunction 

Function check_valve
   ReportColumnOrGunValve valve_status
  if $valve_status == 0
   check_num = 1
   Loop 30 $check_num
      ReportVacuumGauge IGPCO IGPco_status IGPco_pressure
     if $IGPco_status == 4
          log_value = LOG $IGPco_pressure
          IGPco_value = 10.1  *  ( $log_value  ) + 133.6
          if $IGPco_value  < 30
                  SetColumnOrGunValve 1
                  break
          Endif
          Delay 1 min
     endif
    check_num = 1 + $check_num
   EndLoop 
   if $check_num >= 30
       CallFunction Tomo_group_X::reset_all
   endif
 endif
EndFunction 

function TiltZero
   Delay 2
ReportTiltAngle cur_angle

#store stage position
ReportStageXYZ StageX StageY 

#determine defocus
target_focus =  RAND 
target_focus = ( $max_defocus - $min_defocus ) * $target_focus + $min_defocus

#drift and tracking
SetAutofocusOffset 0
SetTargetDefocus $target_focus
G    
G
T
Copy A K
Copy A L

Delay 10
T
if $autoloader == 1
   CallFunction Tomo_group_X::check_autofill
Endif 

Loop $drifttimes index
            AlignTo K
            ReportAlignShift
            dx = $reportedValue3
            dy = $reportedValue4
            dist = sqrt $dx * $dx + $dy * $dy
            rate = $dist / $driftinterval * 10
            echo Rate = $rate A/Sec
            if $rate < $driftcrit
                           echo Drift is low enough after shot $index
                            break 
             Elseif $index < $drifttimes
                     Delay 10
                     T 
              Else
                           echo Drift never got below $driftcrit; Skipping........
                           break
                Endif
Endloop

#store defocus
GoToLowDoseArea R
ReportDefocus
focusplus = $RepVal1
focusmin = $RepVal1
ReportNavItem 
current_index = $reportedValue1
current_X = $reportedValue2
current_Y = $reportedValue3
current_note = $navNote
current_lable = $navLabel
#store image shifts
ReportImageShift
Isxplus = $RepVal1
Isyplus = $RepVal2
Isxminus = $RepVal1
Isyminus = $RepVal2

#acquire tilt image
    IS_X = $Isxplus
    IS_Y = $Isyplus 

SetImageShift $IS_X $IS_Y 1 1
R
ReportLastFrameFile fram_name

#on the fly data processing monitor
test_name @= $fram_name_test
OpenNewFile $test_name 
ReportCurrentFilename 1
fram_path = $reportedValue3
CloseFile $test_name 
RemoveFile $test_name 

tomo_log_name @= $fram_pathtomo_log.txt
Try 
OpenTextFile log_file W 0 $tomo_log_name 
Catch 
OpenTextFile log_file A 0 $tomo_log_name 
EndTry 
WriteLineToFile log_file tomo_start at Nav_Item $current_lable 
FlushTextFile log_file 

#check each point
i = 1
array_dis_ang = {0 }
Loop 50  side_index
    next_index = $current_index + $i
    Try 
        ReportOtherItem $next_index 
        next_X = $reportedValue2
        next_Y = $reportedValue3
        next_note = $navNote
        next_lable = $navLabel
       if $next_note == $current_note
             echo touch next group at $next_index
            Break 
       Endif 
    Catch 
        echo touch next group at $next_index
        Break 
     EndTry 

    stage_X =  ( $next_X - $current_X )
    stage_Y =  ( $next_Y - $current_Y ) * ( cos  $start_angle )

       IS_X  = $stage_X * $IS_xpx + $stage_Y * $IS_xpy + $Isxplus 
       IS_Y = $stage_X * $IS_ypx + $stage_Y * $IS_ypy + $Isyplus   
   
   if $autoloader == 1
     CallFunction Tomo_group_X::check_autofill
  Endif 

    SetImageShift $IS_X $IS_Y 1 1

   Z_defocus = ( $next_Y - $current_Y ) * ( sin $exp_start_angle  ) 

    if $use_focus_forZ == 1
         Z_defocus_toset = $Z_defocus + $focusplus
        SetDefocus $Z_defocus_toset 
        AutoFocus
        ReportDefocus
        Z_diff = ( 1 / ( 1 + $Z_correct  ) ) * ( $reportedValue1 - $focusplus )
     else
        Z_defocus_toset =  ( 1 + $Z_correct  )   * $Z_defocus + $focusplus
       SetDefocus $Z_defocus_toset 
        Z_diff = $Z_defocus
     Endif 

     array_dis_ang = {$array_dis_ang $stage_Y $Z_diff $next_lable }
     echo $array_dis_ang
   #EarlyReturnNextShot 0
    R
    ReportLastFrameFile fram_name
    WriteLineToFile log_file $cur_angle $fram_name $IS_X $IS_Y $next_lable $Z_diff R
   FlushTextFile log_file 

    filename @= mapS_$side_index.mrc
    RemoveFile $filename 
    SaveToOtherFile A MRC CUR $filename 
    
    P_align = { $P_align 0 $Z_diff  }
    M_align = { $M_align 0 $Z_diff  }
    P_align_X = { $P_align_X 0 $stage_X }
    M_align_X = { $M_align_X 0 $stage_X }
     i = $i + 1
EndLoop
group_num = $i - 1
endfunction

function TiltPlus
   
#tilt stage
if $data_mode == 1 or $data_mode == 2
   TiltTo $tiltangle
   GoToLowDoseArea R
Else
   WalkUpTo $tiltangle
   GoToLowDoseArea R
   NormalizeAllLenses 
   delay 2
Endif 
Delay 2
ReportTiltAngle cur_angle
#reset stage XY
MoveStageTo $StageX $StageY
    
#setdefocus and image shift
SetDefocus $focusplus
SetImageShift $Isxplus $Isyplus 1 1

#drift and tracking
T
AlignTo K
ReportAlignShift
dx = $reportedValue3
dy = $reportedValue4
dist = sqrt $dx * $dx + $dy * $dy
echo $dist Alignment 
if $dist > 150
    echo $dist  ClearAlignment 
    ClearAlignment 
    V
    AlignTo H
    V
    Copy A H
    GoToLowDoseArea R
    NormalizeAllLenses 
   delay 3
   T
   AlignTo K
 Endif 
   T
   Copy A K

G
if $autoloader == 1
   CallFunction Tomo_group_X::check_autofill
Endif 
T
Loop $drifttimes index
            AlignTo K
            ReportAlignShift
            dx = $reportedValue3
            dy = $reportedValue4
            dist = sqrt $dx * $dx + $dy * $dy
            rate = $dist / $driftinterval * 10
            echo Rate = $rate A/Sec
            if $rate < $driftcrit
                           echo Drift is low enough after shot $index
                            break 
             Elseif $index < $drifttimes
                         Delay 10
                         T
              Else
                            echo Drift never got below $driftcrit; Skipping........
                             break
                Endif
Endloop

#store defocus
   GoToLowDoseArea R
ReportDefocus focusplus 
#store image shifts
ReportImageShift
Isxplus = $RepVal1
Isyplus = $RepVal2

#record other point
diff_angle = $cur_angle - $start_angle 
j = 2
Loop $group_num side_index
    stage_Y = $array_dis_ang[$j]
     j = $j + 1
    stage_Z = $array_dis_ang[$j]
     j = $j + 1
     next_lable = $array_dis_ang[$j]
     j = $j + 1
    if $plus_num > 7
          filename @= mapP_$side_index.mrc
          ReadOtherFile 0 P $filename 
     Else          
           filename @= mapS_$side_index.mrc
          ReadOtherFile 0 P $filename
       Endif 

        curr_align_num = $side_index * 2
        side_align_X = $P_align_X[$curr_align_num]
        side_align_Y = $P_align[$curr_align_num]
        curr_align_num = $curr_align_num  + 1
       X_diffcorr = $P_align_X[$curr_align_num]
       Z_diffcorr = $P_align[$curr_align_num]

    X_error = 0.001 * $side_align_X
     Z_error = -0.001 * $side_align_Y / ( sin $diff_angle )
     echo Z_error $Z_error um

     if ( $Z_error > 2  OR $Z_error < -2 ) AND $use_focus_forZ == 1
            Z_error  = $Z_error * 0.5
     Endif 
    if $X_error > 2  OR $X_error < -2
          X_error  = $X_error * 0.5
     endif

    X_diffcorr = $X_diffcorr  + $X_error 
     Z_diffcorr = $Z_diffcorr  + $Z_error 

    stage_X_new = $X_diffcorr  
    stage_Y_new =  $stage_Y * ( cos $diff_angle  ) - $Z_diffcorr * ( sin $diff_angle )

     IS_X  = $stage_X_new  * $IS_xpx + $stage_Y_new * $IS_xpy + $Isxplus 
     IS_Y = $stage_X_new  * $IS_ypx + $stage_Y_new * $IS_ypy + $Isyplus 

     Z_defocus = ( 1 + $Z_correct  )  *  ( $stage_Y * ( sin $diff_angle  ) + $Z_diffcorr * ( cos $diff_angle  )  ) + $focusplus

    SetDefocus $Z_defocus

    if $autoloader == 1
       CallFunction Tomo_group_X::check_autofill
    Endif 

   SetImageShift $IS_X $IS_Y 1 1
    #EarlyReturnNextShot 0
     R
    ReportLastFrameFile fram_name
    WriteLineToFile log_file $cur_angle $fram_name $IS_X $IS_Y $next_lable $Z_error R
    FlushTextFile log_file 

    rest_num = MODULO $plus_num 6
    if $rest_num == 5
        filename @= mapP_$side_index.mrc
        RemoveFile $filename 
        SaveToOtherFile A MRC CUR $filename
   Endif 
        AlignTo P 1
        ReportAlignShift
        side_align_X = $reportedValue5
        side_align_Y = $reportedValue6
        curr_align_num = $side_index * 2
        P_align_X[$curr_align_num] = $side_align_X
        P_align[$curr_align_num] = $side_align_Y

         if $M_align[$curr_align_num] == 0.0
                M_align[$curr_align_num] = -1 * $side_align_Y
          Endif 

        curr_align_num = $curr_align_num  + 1
         P_align_X[$curr_align_num] = $X_diffcorr  
         P_align[$curr_align_num] = $Z_diffcorr 


EndLoop
echo record side points
endfunction

Function tiltup_by_step
echo tiltup_by_step
ReportTiltAngle last_angle

tilt_range = $tiltangle - $last_angle
inter_step = 6
SetTiltIncrement $inter_step 
two_inter_step  = 2 * $inter_step 

if $tilt_range > $two_inter_step  
 Loop 100
  Try 1
   TiltUp 
   Catch 
      Delay 1 sec
      TiltUp 
   EndTry 
   ReportTiltAngle last_angle
   tilt_range = $tiltangle - $last_angle 
  if $tilt_range < $inter_step 
   break
  Endif 
 EndLoop 
Endif 

Try 1
   TiltTo $tiltangle 
Catch 
   Delay 1 sec
   TiltTo $tiltangle 
EndTry 
EndFunction  

Function tiltdown_by_step
echo tiltdown_by_step

ReportTiltAngle last_angle

tilt_range = $last_angle - $tiltangle - $Tiltbacklash 
inter_step = 6
SetTiltIncrement $inter_step 
two_inter_step  = 2 * $inter_step 

if $tilt_range > $two_inter_step  
 Loop 100
  Try 1
         TiltDown 
      Catch 
         Delay 1 sec
         TiltDown 
      EndTry 
      ReportTiltAngle last_angle
      tilt_range = $last_angle - $tiltangle + 3
  if $tilt_range < $inter_step 
   break
  Endif 
 EndLoop 
Endif 

tilt_range = $tiltangle + $Tiltbacklash 

Try 1
   TiltTo $tilt_range 
Catch 
   Delay 1 sec
   TiltTo $tilt_range 
EndTry 

Try 1
   TiltTo $tiltangle 
Catch 
   Delay 1 sec
   TiltTo $tiltangle 
EndTry 
EndFunction 

Function check_stage
echo check_stage

if $backgroud_stage == 1
check_times = 15
Loop $check_times check_index
  ReportTiltAngle cur_angle
  diff_A = ABS ( $cur_angle - $tiltangle )
  if $diff_A < 0.5
     break
   elseif $check_index < $check_times 
      Delay 3 sec
   Else
      if $tiltangle  > $cur_angle 
         CallFunction Tomo_group_X::tiltup_by_step
      Else
         CallFunction Tomo_group_X::tiltdown_by_step
        Endif 
   Endif 
EndLoop 
Else
   ReportTiltAngle cur_angle
  diff_A = ABS ( $cur_angle - $tiltangle )
    if $diff_A > 0.5
       if $tiltangle  > $cur_angle 
         CallFunction Tomo_group_X::tiltup_by_step
      Else
         CallFunction Tomo_group_X::tiltdown_by_step
        Endif 
       Endif 
Endif 
CallFunction Tomo_group_X::move_stage
EndFunction 

Function move_stage

ReportStageXYZ cur_stage_x cur_stage_y cur_stage_z
abs_x = ABS ( $cur_stage_x  - $current_X  )
abs_y = ABS ( $cur_stage_y  - $current_Y  )
abs_z = ABS ( $cur_stage_z  - $current_Z  )
if  ( $abs_x  > 0.5 ) or  ( $abs_y  > 0.5 )  or  ( $abs_z > 0.5 )
     bck_current_X = $current_X   +  5
    bck_current_Y = $current_Y   +  5
   MoveStageTo $bck_current_X $bck_current_Y $current_Z 
   MoveStageTo $current_X $current_Y 

   MoveStageTo $current_X $current_Y $current_Z 
#   echo move stage $abs_x  $abs_y  $abs_z 
Endif 
EndFunction
